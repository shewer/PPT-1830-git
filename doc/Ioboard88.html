def delay_time(ms)
  return "\x00\x00\x00"  if ms > 1677216 || ms <0
  return [ms].pack("L").reverse[1,3]
end

def set_output_byte(ti=:T0,ci=:C0)
  return nil if  ci.class!=Symbol || ti.class !=Symbol
  return @sel_test_ch[ci] | @sel_test_board[ti]
end
