def run()
  @test_res=@ch.collect {|ch,tp|
     @ioboard.reset()
    @ioboard.relay_set(@ti,ch)
    [ch,tp.run()]
    }
     @ioboard.reset()
    @jug= @test_res.transpose[1] == [true,true,true,true]
    @jug
    
  
end
