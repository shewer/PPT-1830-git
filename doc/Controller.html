    def run()

    loop {  
  case  @status
    when :Org
      command
         chk_input([:Reset,:Start])

    when :Testing
      command
         chk_input([:Reset])

    when :Tested
      command 
         chk_input([:Reset,:Start,:T1,:T2,:T3,:T4])

  end

}

    end

    def chk_input(input_ar)

    flag=@ioboard.io_status[4] 
input_ar.each{|f|
  return f if ()@input_def[f] & flag)!=0 
} return nil

    end def command(input)

    case input
when :Reset 
  reset()
when :Start
  start()
when :T1,:T2,:T3,:T4
  chk_test_res(input)
end

    end
