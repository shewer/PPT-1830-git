def add_chksum(stri)
  if stri.size==8
      str=stri.clone
      str[7] = 0.chr
    sum=0
    ar=str.bytes
    ar.each {|x| sum += x}
    sum = 256
    str[7]= sum.chr
    return (str)
  end
end

def read_modbus()
  str=""
  begin
    Timeout.timeout(1) do
      loop {
        if (head = @server.getc)== "\x22"
          str << head
          7.times {
            str << @server.getc
          }
          return str
        end  # end if 
        }
    end
      
  rescue Timeout::Error
     return nil
    
  end
  
end
