def read_modbus()
  str=""
  begin
    Timeout.timeout(1) do
      
        if (head = @server.getc)== "\x22"
          str << head
          7.times {
            str << @server.getc
          }
          return str
        end  # end if 
    end

  rescue Timeout::Error
     return Modbus_Error_Code
    
  end
  
end
